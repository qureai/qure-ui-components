# TextBox

export const textBoxArgTypes = {
  label: "your name",
  placeholder: "enter your name",
  type: "text",
  inputMode: "text",
  maxLength: 30,
  id: "text_field",
};

export const argTypes = { inputRef: { table: { disable: true } } };

An input field includes a label and a text field users can type text into.

## Usage

```javascript
import React from "react";
import { InputField } from "@cred/neopop-web/lib/components";

const InputFieldExample = () => {
  return (
    <div>
      <InputField
        type="text"
        label="your name"
        placeholder="enter your name"
        id="text_field"
        autoFocus
      />
    </div>
  );
};

export default InputFieldExample;
```
