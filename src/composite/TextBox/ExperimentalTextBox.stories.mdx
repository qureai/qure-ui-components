## Usage

```javascript
import React from "react";
import { TextBox } from "@qure-ui-components";

const TexBoxExample = () => {
  return (
    <TextBox
      type="text"
      label="your name"
      placeholder="enter your name"
      id="text_field"
    />
  );
};

export default TexBoxExample;
```
