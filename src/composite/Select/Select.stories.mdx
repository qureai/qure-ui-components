## Usage

```javascript
import React from 'react';
import Select from '@qure-ui-components/Select'

export default function () {
  const [value, setValue] = React.useState()
  const options = [
    {
      label: "Apple",
      value: "apple",
    },
    {
      label: "Orange",
      value: "orange",
    },
    {
      label: "Tomato",
      value: "tomato",
    },
  ];
  const handleChange = (e: React.ChangeEvent<any>) => {
    setValue(((e.target) as HTMLInputElement).value)
  }
  return (
    <Select name="fruits" label="Fruits" options={options} handleChange={handleChange} />
  )
}
```

## Props

<div style={{overflowX: 'auto'}}>

| prop         | description                                      | type                                  |
| ------------ | ------------------------------------------------ | ------------------------------------- |
| name         | string passed to name attribute                  | string                                |
| label        | string that shows up as label for the select tag | string                                |
| defaultValue | default value to show on first render            | string                                |
| options      | An array of options                              | `{label: string, value: string} []`   |
| handleChange | fn that is called on selecting an option         | `(e: React.ChangeEvent<any>) => void` |

</div>
